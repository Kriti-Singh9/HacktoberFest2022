18. Print Pascal's triangle. 
   1
    1 1
    1 2 1
    1 3 3 1
    1 4 6 4 1
    
    int main()
{
    int r, c = 1;
    cin >> r ;


    for(int i = 0; i < r ; i++)
    {
        
        for(int j = 0; j <= i; j++)
        {
            if (j == 0 || i == 0)
                c = 1;
            else
                c = c * (i-j+1)/j;


            cout << c << "   ";
        }
        cout << endl;
    }


    return 0;
}
